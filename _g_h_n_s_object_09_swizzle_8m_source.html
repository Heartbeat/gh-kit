00049 BOOL GH_PerformSwizzle(Class klass, SEL origSel, SEL altSel, BOOL forInstance) {
00050     
00051     // Make sure the class isn't nil
00052     if (klass == nil)
00053         return NO;
00054     
00055     // Look for the methods in the implementation of the immediate class
00056     Class iterKlass = (forInstance ? klass : klass->isa);
00057     Method origMethod = NULL, altMethod = NULL;
00058     unsigned int methodCount = 0;
00059     Method *mlist = class_copyMethodList(iterKlass, &methodCount);
00060     if ( mlist != NULL ) {
00061         int i;
00062         for (i = 0; i < methodCount; ++i) {
00063             if ( method_getName(mlist[i]) == origSel )
00064                 origMethod = mlist[i];
00065             if (method_getName(mlist[i]) == altSel)
00066                 altMethod = mlist[i];
00067         }
00068     }
00069     
00070     // if origMethod was not found, that means it is not in the immediate class
00071     // try searching the entire class hierarchy with class_getInstanceMethod
00072     // if not found or not added, bail out
00073     if ( origMethod == NULL ) {
00074         origMethod = class_getInstanceMethod(iterKlass, origSel);
00075         if ( origMethod == NULL )
00076             return NO;
00077         if ( class_addMethod(iterKlass, method_getName(origMethod), method_getImplementation(origMethod), method_getTypeEncoding(origMethod)) == NO )
00078             return NO;
00079     }
00080     
00081     // same thing with altMethod
00082     if ( altMethod == NULL ) {
00083         altMethod = class_getInstanceMethod(iterKlass, altSel);
00084         if ( altMethod == NULL ) 
00085             return NO;
00086         if ( class_addMethod(iterKlass, method_getName(altMethod), method_getImplementation(altMethod), method_getTypeEncoding(altMethod)) == NO )
00087             return NO;
00088     }
00089     
00090     //clean up
00091     free(mlist);
00092     
00093     // we now have to look up again for the methods in case they were not in the class implementation,
00094     //but in one of the superclasses. In the latter, that means we added the method to the class,
00095     //but the Leopard APIs is only 'class_addMethod', in which case we need to have the pointer
00096     //to the Method objects actually stored in the Class structure (in the Tiger implementation, 
00097     //a new mlist was explicitely created with the added methods and directly added to the class; 
00098     //thus we were able to add a new Method AND get the pointer to it)
00099     
00100     // for simplicity, just use the same code as in the first step
00101     origMethod = NULL;
00102     altMethod = NULL;
00103     methodCount = 0;
00104     mlist = class_copyMethodList(iterKlass, &methodCount);
00105     if ( mlist != NULL ) {
00106         int i;
00107         for (i = 0; i < methodCount; ++i) {
00108             if ( method_getName(mlist[i]) == origSel )
00109                 origMethod = mlist[i];
00110             if (method_getName(mlist[i]) == altSel)
00111                 altMethod = mlist[i];
00112         }
00113     }
00114     
00115     // bail if one of the methods doesn't exist anywhere
00116     // with all we did, this should not happen, though
00117     if (origMethod == NULL || altMethod == NULL)
00118         return NO;
00119     
00120     // now swizzle
00121     method_exchangeImplementations(origMethod, altMethod);
00122     
00123     //clean up
00124     free(mlist);
00125     
00126     return YES;
00127 }
