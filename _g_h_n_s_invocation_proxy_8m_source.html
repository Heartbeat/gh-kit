00120 - (void)forwardInvocation:(NSInvocation *)invocation {  
00121     self.invocation = invocation;
00122     
00123     invocation.target = target_;
00124     // For argument proxy, if we are overriding the invoking selector
00125     if (selector_ != NULL) invocation.selector = selector_;
00126     
00127     [invocation retainArguments];
00128     
00129     // Check for unsupported combinations
00130     if (thread_ && delay_ >= 0)
00131         [NSException raise:NSInvalidArgumentException format:@"Running on thread with delay is not supported at the same time"];
00132     if (thread_ && detachCallback_)
00133         [NSException raise:NSInvalidArgumentException format:@"Running on thread with detached callback is not supported at the same time"];
00134     if (detachCallback_ && delay_ >= 0)
00135         [NSException raise:NSInvalidArgumentException format:@"Running with delay and detached callback is not supported at the same time"];
00136     
00137     [delegate_ proxy:self willInvoke:invocation];
00138     if (thread_) {
00139         [invocation_ performSelector:@selector(invoke) onThread:thread_ withObject:nil waitUntilDone:waitUntilDone_];
00140     } else if (detachCallback_) {       
00141         [NSThread detachNewThreadSelector:@selector(invoke:) toTarget:detachCallback_ withObject:invocation_];
00142     } else {
00143         if (delay_ >= 0) {
00144             [invocation_ performSelector:@selector(invoke) withObject:nil afterDelay:delay_];
00145         } else {
00146             [invocation_ performSelector:@selector(invoke) withObject:nil];
00147         }
00148     }
