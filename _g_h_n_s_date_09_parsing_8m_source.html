00072 + (NSDate *)gh_parseTimeSinceEpoch:(id)timeSinceEpoch withDefault:(id)value offsetForTimeZone:(NSTimeZone *)timeZone {
00073   if (!timeSinceEpoch) return value;
00074     NSDate *normalDate = [NSDate dateWithTimeIntervalSince1970:[timeSinceEpoch doubleValue]];
00075   if (!timeZone)
00076     return normalDate;
00077   NSTimeZone *localTimeZone = [NSTimeZone localTimeZone];
00078   if ([localTimeZone isEqualToTimeZone:timeZone])
00079     return normalDate;
00080   // The following adapted from http://stackoverflow.com/questions/1081647/how-to-convert-time-to-the-timezone-of-the-iphone-device/1082179#1082179
00081   NSInteger offset = [timeZone secondsFromGMTForDate:normalDate];
00082   NSInteger localOffset = [localTimeZone secondsFromGMTForDate:normalDate];
00083   NSTimeInterval difference = localOffset - offset;
00084   return [[[NSDate alloc] initWithTimeInterval:difference sinceDate:normalDate] autorelease];
00085 }
