00065 + (NSDate *)gh_parseTimeSinceEpoch:(id)timeSinceEpoch withDefault:(id)value offsetForTimeZone:(NSTimeZone *)timeZone {
00066   if (!timeSinceEpoch) return value;
00067     NSDate *normalDate = [NSDate dateWithTimeIntervalSince1970:[timeSinceEpoch doubleValue]];
00068   if (!timeZone)
00069     return normalDate;
00070   NSTimeZone *localTimeZone = [NSTimeZone localTimeZone];
00071   if ([localTimeZone isEqualToTimeZone:timeZone])
00072     return normalDate;
00073   // The following adapted from http://stackoverflow.com/questions/1081647/how-to-convert-time-to-the-timezone-of-the-iphone-device/1082179#1082179
00074   NSInteger offset = [timeZone secondsFromGMTForDate:normalDate];
00075   NSInteger localOffset = [localTimeZone secondsFromGMTForDate:normalDate];
00076   NSTimeInterval difference = localOffset - offset;
00077   return [[[NSDate alloc] initWithTimeInterval:difference sinceDate:normalDate] autorelease];
