00029 /* blk0() and blk() perform the initial expand. */
00030 /* I got the idea of expanding during the round function from SSLeay */
00031 #ifdef __LITTLE_ENDIAN__
00032 #define blk0(i) (block->l[i] = (rol(block->l[i],24)&0xFF00FF00) \
00033 |(rol(block->l[i],8)&0x00FF00FF))
00034 #else
00035 #define blk0(i) block->l[i]
00036 #endif
00037 #define blk(i) (block->l[i&15] = rol(block->l[(i+13)&15]^block->l[(i+8)&15] \
00038 ^block->l[(i+2)&15]^block->l[i&15],1))

00040 /* (R0+R1), R2, R3, R4 are the different operations used in SHA1 */
00041 #define R0(v,w,x,y,z,i) z+=((w&(x^y))^y)+blk0(i)+0x5A827999+rol(v,5);w=rol(w,30);
00042 #define R1(v,w,x,y,z,i) z+=((w&(x^y))^y)+blk(i)+0x5A827999+rol(v,5);w=rol(w,30);
00043 #define R2(v,w,x,y,z,i) z+=(w^x^y)+blk(i)+0x6ED9EBA1+rol(v,5);w=rol(w,30);
00044 #define R3(v,w,x,y,z,i) z+=(((w|x)&y)|(w&x))+blk(i)+0x8F1BBCDC+rol(v,5);w=rol(w,30);
00045 #define R4(v,w,x,y,z,i) z+=(w^x^y)+blk(i)+0xCA62C1D6+rol(v,5);w=rol(w,30);

00120 void SHA1Update(SHA1_CTX* context, unsigned char* data, unsigned int len)
00121 {
00122   unsigned int i, j;
00123   
00124   j = (unsigned int)((context->count[0] >> 3) & 63);
00125   if ((context->count[0] += len << 3) < (len << 3)) context->count[1]++;
00126   context->count[1] += (len >> 29);
00127   if ((j + len) > 63) {
00128     memcpy(&context->buffer[j], data, (i = 64-j));
00129     SHA1Transform(context->state, context->buffer);
00130     for ( ; i + 63 < len; i += 64) {
00131       SHA1Transform(context->state, &data[i]);
00132     }
00133     j = 0;
00134   }
00135   else i = 0;
00136   memcpy(&context->buffer[j], &data[i], len - i);
